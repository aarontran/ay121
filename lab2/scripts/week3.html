nums = np.arange(-4,4)
numzeros = 56

coeffs1 = np.sinc(5./8*nums)
coeffs2 = np.sinc(5./8*nums/np.pi)
coeffs1 = np.append(coeffs1, np.zeros(numzeros))
coeffs2 = np.append(coeffs2, np.zeros(numzeros))

fsamp = 200e6
nsamp = nums.size + numzeros
times = np.arange(nsamp)/fsamp * 1e6  # Microseconds
freqs = np.fft.fftfreq(nsamp, 1./fsamp) / 1e6  # MHz
freqs = np.roll(freqs, freqs.size/2)

freq_resp_1 = np.roll(np.fft.fft(coeffs1), freqs.size/2)
freq_resp_2 = np.roll(np.fft.fft(coeffs2), freqs.size/2)
freq_resp_ideal_1 = [1.6 if abs(f) < 62.5 else 0 for f in freqs]
freq_resp_ideal_2 = [1.6*np.pi if abs(f) < 5/(8*np.pi)*100 else 0 for f in freqs]

plt.figure(figsize=(4,3))
plt.plot(freqs, np.abs(freq_resp_1), ':k.')
plt.plot(freqs, np.abs(freq_resp_2), ':r.')
plt.plot(freqs, freq_resp_ideal_1, '-b.')
plt.plot(freqs, freq_resp_ideal_2, '-g.')
plt.xlabel(r'Frequency ($\mathrm{MHz}$)', fontsize=14)

    data_imag[i] = np.fromfile('../data-week3/ddc_imag_bram_'+str(i), dtype='>i4')
